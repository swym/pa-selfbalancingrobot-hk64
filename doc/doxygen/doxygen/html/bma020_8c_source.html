183 void bma020_read_raw_acceleration(acceleration_t* raw_a_vector)
184 {
185  uint8_t temp_data;
186 
187  twi_send_buffer[0] = BMA020_REGISTER_VALUE_X_LSB;
188 
189  twi_master_set_ready();
190  twi_send_data(BMA020_TWI_ADDRESS, 1);
191  twi_receive_data(BMA020_TWI_ADDRESS, 6);
192 
193  temp_data = twi_receive_buffer[0];
194  raw_a_vector->x = (uint16_t)((temp_data & 0xC0));
195  temp_data = twi_receive_buffer[1];
196  raw_a_vector->x = (uint16_t)(raw_a_vector->x | temp_data << 8);
197 
198  temp_data = twi_receive_buffer[2];
199  raw_a_vector->y = (uint16_t)((temp_data & 0xC0));
200  temp_data = twi_receive_buffer[3];
201  raw_a_vector->y = (uint16_t)(raw_a_vector->y | temp_data << 8);
202 
203  temp_data = twi_receive_buffer[4];
204  raw_a_vector->z = (uint16_t)((temp_data & 0xC0));
205  temp_data = twi_receive_buffer[5];

765 {
766  if(counter == 0 || counter == 1 || counter == 2 || counter == 3) {
767 
768  uint8_t register_value;

777
